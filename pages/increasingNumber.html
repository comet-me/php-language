Ascending Number Pattern

$n = 6; // This $n here will determine the size. For instance this will print 5 rows
for($i = 1; $i <= $n; $i++){ // This will create 5 rows
for($j = 1; $j < $i; $j++){ // Cycle One 0x: 1<1 Cycle Two 1x: 1<2 Cycle Three 2x: 1<3,2<3
// Cycle Four 3x: 1<4,2<4,3<4, Cycle Five 4x: 1<5,2<5,3<5,4<5 Cycle Six 5x: 1<6,2<6,...
print $j."  "; //cycle two: 1*1 cycle three: 1*1,1*2
//cycle four: 1*1,1*2,1*3 cycle five: 1*1,1*2,1*3,1*4 Cycle Six: 1*1,1*2,...
}
print "<br>";
}
